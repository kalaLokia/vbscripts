Function RowCount(RowName as String)
   Dim SearchCell As Range 
   With Worksheets("BONG") 
    If Not Application.IsNA(Application.MATCH(RowName, Range("B:B"), 0)) 
      Then 
        Set SearchCell = Cells(Application.MATCH(RowName, Range("B:B"), 0), 2) 
        RowCount = SearchCell.MergeArea.Rows.Count 
    End If 

   End With 

End Function
